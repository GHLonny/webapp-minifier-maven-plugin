37     public MinifiedFileBuilder(final File targetDirectory, final String prefix,
38           final String extension) {
39        if (targetDirectory == null) {
40           throw new IllegalArgumentException(
41                 "The target directory cannot be null");
42        }
43        if (prefix == null) {
44           throw new IllegalArgumentException("The prefix cannot be null");
45        }
46        if (extension == null) {
47           throw new IllegalArgumentException("The file extension cannot be null");
48        }
49        this.targetDirectory = targetDirectory;
50        this.prefix = prefix;
51        this.extension = extension;
52     }
53  
54     /**
55      * Indicates if the next call to {@link #getCurrentFile()} will return a new
56      * file.
57      * 
58      * @return <code>true</code> if the next request for a
59      *         {@link MinifiedFileInfo} will be a new instance.
60      */
61     public boolean isNewFile() {
62        return this.currentFile == null;
63     }
64  
65     /**
66      * Returns the currently minified file.
67      * 
68      * @return the currently minified file.
69      */
70     public MinifiedFileInfo getCurrentFile() {
71        if (isNewFile()) {
72           final File file = new File(this.targetDirectory, this.prefix + '-'
73                 + ++this.fileCount + '.' + this.extension);
74           this.currentFile = new MinifiedFileInfo(file);
75        }
76        return this.currentFile;
77     }
