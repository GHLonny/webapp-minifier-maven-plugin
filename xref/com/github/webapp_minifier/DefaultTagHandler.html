85        final MinifiedFileBuilder jsFileBuilder = new MinifiedFileBuilder(
86              options.getTargetDirectory(), options.getJsPrefix(), "js");

88              createJavaScriptCompressor(options), jsFileBuilder);

161    protected Compressor createJavaScriptCompressor(final OverridablePluginOptions options) {
162       final Compressor compressor;
163       if (options.isSkipJsMinify()) {
164          compressor = null;
165       } else {
166          switch (options.getJsCompressorEngine()) {
167          case CLOSURE:
168             compressor = createClosureJsCompressor(options);

172             compressor = createYuiJsCompressor(options);

186    @Override
187    public boolean handleComment(final String text) throws ParseOptionException {
188       this.log.debug("Handling comment '" + text + "'");
189 
190       if (this.optionsParser.containsOptionsHeader(text)) {
191          final InlineConfigurationHandler defaultInlineConfigurationHandler = new DefaultInlineConfigurationHandler(
192                this.options);
193          this.optionsParser.parse(text, defaultInlineConfigurationHandler, this);
194          this.cssContext.setCompressor(createCssCompressor(this.options));
195          this.jsContext.setCompressor(createJavaScriptCompressor(this.options));
196          this.jsContext.setMinifier(this.options.getJsCompressorEngine().toString());
197          return true;
198       }
199       return false;

294    /**
295     * Minifies the external code.
296     *
297     * @param urlString
298     *           the URL to the external code.
299     * @param compressor
300     *           the compressor to use.
301     * @param builder
302     *           the builder for the compressed files.
303     * @return the original URL or the file name containing the minified code.
304     * @throws FileNotFoundException
305     *            if the minified file could not be created.
306     * @throws IOException
307     *            if any other error occurred while writing to the minified file.
308     */
309    protected String minifyExternalCode(final String urlString, final MinificationContext context)
310          throws FileNotFoundException, IOException {
311       String result = urlString;
312       final Compressor compressor = context.getCompressor();
313       final MinifiedFileBuilder builder = context.getFileBuilder();
314       if (compressor != null) {
315          // TODO: Determine if the file has already been minified.
316          File sourceFile = new File(this.pluginOptions.getTargetDirectory(), urlString);
317 
318          // If the file exists, get it's input stream.
319          InputStream inputStream = null;
320          if (sourceFile.exists()) {
321             inputStream = new FileInputStream(sourceFile);
322          } else {
323             // Otherwise, search the "other directories" for a matching
324             // file.
325             for (final Entry<Object, Object> entry : this.pluginOptions.getOtherDirectories()
326                   .entrySet()) {
327                String key = (String) entry.getKey();
328                if (!key.endsWith("/")) {
329                   key = key + '/';
330                }
331                if (urlString.startsWith(key)) {
332                   this.log.debug("Searching for '" + urlString + "' using key '" + key
333                         + "' which points to '" + entry.getValue() + "'");
334                   final File otherTarget = new File((String) entry.getValue());
335                   sourceFile = new File(otherTarget, urlString.substring(key.length()));
336                   if (sourceFile.exists()) {
337                      inputStream = new FileInputStream(sourceFile);
338                      break;
339                   }
340                }
341             }
342          }
343 
344          // If an input stream was not found for the current URL string, skip
345          // the tag and create a new minified file next time.
346          if (inputStream == null) {
347             this.log.debug("Did not find '" + urlString + "'.  Its content will not be minified.");
348             builder.finishFile();
349          } else {
350             // Minify the contents of the input stream.
351             final String original = IOUtil.toString(inputStream, this.pluginOptions.getEncoding(),
352                   8192);
353             final MinifiedFileInfo fileInfo = builder.getCurrentFile();
354             final File destinationFile = fileInfo.getFile();
355             final MinifiedFileMetrics metrics = new MinifiedFileMetrics();
356             metrics.setSource(urlString);
357             metrics.setMinifier(context.getMinifier());
358             result = minify(compressor, original, destinationFile, metrics);
359             this.currentFile.getMinifiedFiles().add(metrics);
360 
361             // Keep track of which inputs went into which outputs.
362             fileInfo.getFiles().add(sourceFile);
363             if (this.minifiedFiles.containsKey(sourceFile)) {
364                this.minifiedFiles.get(sourceFile).add(fileInfo);
365             } else {
366                final Collection<MinifiedFileInfo> c = new ArrayList<MinifiedFileInfo>();
367                c.add(fileInfo);
368                this.minifiedFiles.put(sourceFile, c);
369             }
370          }
371       }
372       return result;
373    }

375    /**
376     * Minifies the embedded CSS or JavaScript.
377     *
378     * @param text
379     *           the CSS or JavaScript to minify.
380     * @param skipEmbedded
381     *           indicates if minification should be skipped.
382     * @param mergeEmbedded
383     *           indicates if the embedded code should be merged with an external file.
384     * @param compressor
385     *           the compressor to use.
386     * @param builder
387     *           the builder for the compressed files.
388     * @param metrics
389     *           the place to store metrics about minification.
390     * @return the minified result or <code>null</code> if it was added to another file.
391     * @throws FileNotFoundException
392     *            if the minified file could not be created.
393     * @throws IOException
394     *            if any other error occurred while writing to the minified file.
395     */
396    protected String minifyEmbedded(final String text, final boolean skipEmbedded,
397          final boolean mergeEmbedded, final MinificationContext context,
398          final MinifiedFileMetrics metrics) throws FileNotFoundException, IOException {
399       final String result;
400       final Compressor compressor = context.getCompressor();
401       final MinifiedFileBuilder builder = context.getFileBuilder();
402       if (skipEmbedded || (compressor == null)) {
403          result = text;
404          builder.finishFile();
405       } else if (mergeEmbedded && !builder.isNewFile()) {
406          final MinifiedFileInfo fileInfo = builder.getCurrentFile();
407          final File file = fileInfo.getFile();
408          metrics.setMinifier(context.getMinifier());
409          minify(compressor, text, file, metrics);
410          fileInfo.setEmbeddedIncluded(true);
411          result = null;
412       } else {
413          final long startTime = System.nanoTime();
414          result = compressor.compress(text);
415          final long endTime = System.nanoTime();
416          metrics.setTime(endTime - startTime);
417          metrics.setOriginalLength(text.length());
418          metrics.setMinifiedLength(result.length());
419          metrics.setMinifier(context.getMinifier());
420          builder.finishFile();
421       }
422       return result;
423    }

425    /**
426     * Minifies the input using the provided compressor.
427     *
428     * @param compressor
429     *           the compressor
430     * @param input
431     *           the input to be minified.
432     * @param destinationFile
433     *           the destination file.
434     * @param metrics
435     *           the place to store metrics about minification.
436     * @return the destination file name if the file was created or <code>null</code> if the minified
437     *         output was appended to the file.
438     * @throws FileNotFoundException
439     *            if the file exists but is a directory rather than a regular file, does not exist
440     *            but cannot be created, or cannot be opened for any other reason.
441     * @throws IOException
442     *            if an error occurs while copying the minified output to the file.
443     */
444    protected String minify(final Compressor compressor, final String input,
445          final File destinationFile, final MinifiedFileMetrics metrics)
446                throws FileNotFoundException, IOException {
447       String result;
448       OutputStream oStream = null;
449       try {
450          if (destinationFile.exists()) {
451             result = null;
452          } else {
453             result = destinationFile.getName();
454          }
455          oStream = new BufferedOutputStream(new FileOutputStream(destinationFile, true));
456          final long startTime = System.nanoTime();
457          final String compressed = compressor.compress(input);
458          final long endTime = System.nanoTime();
459          metrics.setTime(endTime - startTime);
460          metrics.setOriginalLength(input.length());
461          metrics.setMinifiedLength(compressed.length());
462          metrics.setDestination(destinationFile.getName());
463          IOUtil.copy(compressed, oStream);
464          this.log.info("Reduced input from " + input.length() + " to " + compressed.length()
465                + " characters");
466       } finally {
467          IOUtil.close(oStream);
468       }
469       return result;
470    }
